git config --list - просмотр текущей информации о настройках гит
git config --global user.name "Roman Komaldinov" - задание имени пользователя
git config --global user.email r.komaldinov@mail.ru
git init - определить репозиторий гит в текущей папке
ls -la - просмотр текущего каталога включая скрытые объекты
git branch - просмотр названия текущей ветки
git branch -m <name> - изменение имени текущей ветки
git add <name> - подготовить файл для сохранения в гит
git add . - подготовить всей файлы для сохранения в гит
git commit - сохранить подготовленные файлы в гит
git checkout - выбрать версию и выгрузить ее из репозитория в рабочую директорию (полная перезапись директории).

git branch - список всех веток
git branch -m <new name> - переименование текущеф ветки
git branch <name> - создание новой ветки
git checkout <name> - переход в указанную ветку или коммит
git checkout -b <new name> - создание новой ветки и переход в нее
git log - отображение коммитов в текущей ветке
git branch -d <branch name> - удалить ветку

git merge <feature branch name> - слияние указанной ветки с текущей веткой
